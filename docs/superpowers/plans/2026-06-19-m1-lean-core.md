# MDflow M1 — Lean Core Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a standalone, MIT-licensed MDflow desktop app with an open/save markdown editor, a live preview, view-mode switching, hotkeys, a refined dark theme, and the auto-updater plugin installed but dormant.

**Architecture:** Tauri 2 + Rust native shell. Frontend is plain TypeScript modules (no framework) wired by a thin `main.ts`. Each module owns one responsibility: `editor.ts` (CodeMirror 6), `preview.ts` (markdown-it render), `views.ts` (layout/view modes), `files.ts` (IPC open/save), `state.ts` (persisted UI state). Rust side exposes file IO + word count as Tauri commands.

**Tech Stack:** Tauri 2, Rust, Vite, TypeScript, CodeMirror 6, markdown-it, highlight.js, Vitest (pure functions only). Package manager: **npm**.

export type ViewMode = 'split' | 'editor' | 'preview';
export type UIState = { viewMode: ViewMode; zoom: number };

const KEY = 'mdflow.ui';
const DEFAULTS: UIState = { viewMode: 'split', zoom: 1 };

export function loadState(): UIState {
  try {
    const raw = localStorage.getItem(KEY);
    if (!raw) return { ...DEFAULTS };
    const parsed = JSON.parse(raw);
    return { ...DEFAULTS, ...parsed };
  } catch {
    return { ...DEFAULTS };
  }
}

export function saveState(s: UIState): void {
  localStorage.setItem(KEY, JSON.stringify(s));
}

### Task 7: editor.ts — CodeMirror 6 editor

**Files:**
- Create: `src/editor.ts`

**Interfaces:**
- Consumes: the `#editor` element from Task 3.
- Produces:
  - `createEditor(parent: HTMLElement, onChange: (doc: string) => void): EditorHandle`
  - `type EditorHandle = { getDoc(): string; setDoc(text: string): void }`
  - Markdown highlighting + soft-wrap enabled; calls `onChange` on every doc change.

- [ ] **Step 1: Implement `src/editor.ts`**

```ts
import { EditorView, keymap, lineNumbers } from '@codemirror/view';
import { EditorState } from '@codemirror/state';
import { defaultKeymap, history, historyKeymap } from '@codemirror/commands';
import { markdown } from '@codemirror/lang-markdown';
import { languages } from '@codemirror/language-data';

export type EditorHandle = { getDoc(): string; setDoc(text: string): void };

export function createEditor(parent: HTMLElement, onChange: (doc: string) => void): EditorHandle {
  const view = new EditorView({
    parent,
    state: EditorState.create({
      doc: '',
      extensions: [
        lineNumbers(),
        history(),
        keymap.of([...defaultKeymap, ...historyKeymap]),
        markdown({ codeLanguages: languages }),
        EditorView.lineWrapping,
        EditorView.updateListener.of((u) => { if (u.docChanged) onChange(u.state.doc.toString()); }),
      ],
    }),
  });
  return {
    getDoc: () => view.state.doc.toString(),
    setDoc: (text) => view.dispatch({ changes: { from: 0, to: view.state.doc.length, insert: text } }),
  };
}
```

### Task 8: files.ts — IPC open/save + dialogs

**Files:**

**Interfaces:**
- Consumes: Rust commands from Task 6; `@tauri-apps/plugin-dialog`.
- Produces:

  - `openFile(): Promise<OpenResult>` — dialog filtered to md/markdown/txt → reads via `read_file`.
  - `saveFile(path: string | null, contents: string): Promise<string | null>` — if `path` is null, save dialog; writes via `save_file`; returns the path written (or null if cancelled).
  - `getInitialFile(): Promise<OpenResult>` — resolves the launch-arg file if any.

- [ ] **Step 1: Implement `src/files.ts`**

```ts
import { invoke } from '@tauri-apps/api/core';
import { open, save } from '@tauri-apps/plugin-dialog';

export type OpenResult = { path: string; contents: string } | null;

const FILTERS = [{ name: 'Markdown', extensions: ['md', 'markdown', 'txt'] }];

export async function openFile(): Promise<OpenResult> {
  const path = await open({ multiple: false, filters: FILTERS });
  if (typeof path !== 'string') return null;
  const contents = await invoke<string>('read_file', { path });
  return { path, contents };
}

export async function saveFile(path: string | null, contents: string): Promise<string | null> {
  let target = path;
  if (!target) {
    target = await save({ filters: FILTERS });
    if (!target) return null;
  }
  await invoke('save_file', { path: target, contents });
  return target;
}

export async function getInitialFile(): Promise<OpenResult> {
  const path = await invoke<string | null>('get_initial_file');
  if (!path) return null;
  const contents = await invoke<string>('read_file', { path });
  return { path, contents };
}
```

**Files:**

**Interfaces:**

- [ ] **Step 1: Implement `src/main.ts`**

```ts
import './styles.css';
import { createEditor } from './editor';
import { renderMarkdown } from './preview';
import { openFile, saveFile, getInitialFile } from './files';
import { applyViewMode, applyZoom } from './views';
import { loadState, saveState, type ViewMode } from './state';
import { invoke } from '@tauri-apps/api/core';

const previewEl = document.getElementById('preview')!;
const statusPath = document.getElementById('status-path')!;
const statusWords = document.getElementById('status-words')!;

let currentPath: string | null = null;
let ui = loadState();

const editor = createEditor(document.getElementById('editor')!, schedulePreview);

let timer: number | undefined;
function schedulePreview(doc: string): void {
  clearTimeout(timer);
  timer = window.setTimeout(() => updatePreview(doc), 300);
}

async function updatePreview(doc: string): Promise<void> {
  previewEl.innerHTML = renderMarkdown(doc);
  const n = await invoke<number>('word_count', { text: doc });
  statusWords.textContent = `${n} words`;
}

function setMode(mode: ViewMode): void {
  ui = { ...ui, viewMode: mode };
  applyViewMode(mode);
  saveState(ui);
}

async function doOpen(): Promise<void> {
  const r = await openFile();
  if (!r) return;
  currentPath = r.path;
  editor.setDoc(r.contents);
  statusPath.textContent = r.path;
  updatePreview(r.contents);
}

async function doSave(): Promise<void> {
  const written = await saveFile(currentPath, editor.getDoc());
  if (written) { currentPath = written; statusPath.textContent = written; }
}

document.getElementById('btn-open')!.addEventListener('click', doOpen);
document.getElementById('btn-save')!.addEventListener('click', doSave);
document.getElementById('mode-split')!.addEventListener('click', () => setMode('split'));
document.getElementById('mode-editor')!.addEventListener('click', () => setMode('editor'));
document.getElementById('mode-preview')!.addEventListener('click', () => setMode('preview'));

window.addEventListener('keydown', (e) => {
  if (!(e.metaKey || e.ctrlKey)) return;
  const k = e.key.toLowerCase();
  const map: Record<string, () => void> = {
    o: doOpen, s: doSave,
    p: () => setMode('preview'), e: () => setMode('editor'), b: () => setMode('split'),
  };
  if (map[k]) { e.preventDefault(); map[k](); }
});

// init
applyViewMode(ui.viewMode);
applyZoom(ui.zoom);
getInitialFile().then((r) => {
  if (!r) return;
  currentPath = r.path;
  editor.setDoc(r.contents);
  statusPath.textContent = r.path;
  updatePreview(r.contents);
});
```
